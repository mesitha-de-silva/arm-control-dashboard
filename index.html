[~,~,Vd] = svd(D,'econ'); th = Vd(:,end); Qm = [th(1) th(4) th(5); th(4) th(2) th(6); th(5) th(6) th(3)]; uv = th(7:9); s = th(10); ev = eig((Qm+Qm.')/2); if all(ev<0), Qm=-Qm; uv=-uv; s=-s; ev=-ev; end ok = all(ev>0);
